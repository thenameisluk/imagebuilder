_Quick installation later_

setting auto flash to manual
and main partition to /dev/mmcblk1p4

partitions
/boot p13
/ p14
swap p15

mkfs -t ext4 -O ^has_journal -m 0 -L bootemmc /dev/mmcblk1p13
mkfs -t btrfs -m single -L rootemmc /dev/mmcblk1p14
mount -o ssd,compress-force=zstd,noatime,nodiratime /dev/mmcblk1p14 /mnt
mkdir -p /mnt/boot
mount /dev/mmcblk1p13 /mnt/boot
mkswap /dev/mmcblk1p15

rsync -axADHSX --no-inc-recursive --delete /boot/ /mnt/boot
rsync -axADHSX --no-inc-recursive --delete --exclude='/swap/*' / /mnt

fixup the labels

sed -i 's,bootpart,bootemmc,g;s,rootpart,rootemmc,g' /mnt/etc/fstab

i needed to chroot into it to generate the kpart

chroot /mnt mount -t proc proc /proc 
chroot /mnt mount -t sysfs sysfs /sys

export PS1="(chroot) ${PS1}"
